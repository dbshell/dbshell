====== Export text fields ======
This script exports text column named ''Description'' from table ''Genre''. For each row is created separate file
names ''genre${GenreId}.txt'' , where ''${GenreId}'' is replaced with value of GenreId column.

<code xml>
<?xml version="1.0" encoding="utf-8" ?>
<Batch
    xmlns="http://schemas.dbshell.com/core"
    >
    <ForEach Source="{Query 'SELECT GenreId, Description FROM Genre'}">
        <SaveToFile File="genre${GenreId}.txt" Value="${Description}" />
    </ForEach>
</Batch>    
</code>
