====== Transform CSV file to CDL ======

This script transforms CSV file (with default settings) into binary CDL file.

<code xml>
<?xml version="1.0" encoding="utf-8" ?>
<Batch
    xmlns="http://schemas.dbshell.com/core"
    xmlns:dbmouse="http://schemas.dbmouse.com/dbshell"
>
  <CopyTable>
    <CopyTable.Source>
    	<CsvFile Name='c:/test/genre.csv' />
    </CopyTable.Source>
    <CopyTable.Target>
    	<CdlFile Name='c:/test/genre.cdl' />
    </CopyTable.Target>
  </CopyTable>
</Batch>
</code>
