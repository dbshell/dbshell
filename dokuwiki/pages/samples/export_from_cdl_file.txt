====== Export from CDL file ======

This example exports table Genre into dst1.sql, using intermediate file genre.cdl.

<code xml>
<?xml version="1.0" encoding="utf-8" ?>
<Batch
    xmlns="http://schemas.dbshell.com/core"
    xmlns:ds="http://schemas.dbshell.com/dataset"
>
	<CopyTable Source="{Table Genre}" Target="{File 'genre.cdl'}" />

	<ds:DataSet />
	<ds:LoadFile Table='Genre' File='genre.cdl' />
	<ds:Prepare />
	<ds:WriteSql File='ds1.sql' />	
</Batch>
</code>
