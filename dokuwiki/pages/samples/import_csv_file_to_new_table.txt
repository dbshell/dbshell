====== Import CSV file to new table ======

This scripts creates table GenreFromFile with columns from CSV file genre.csv and imports data into this table.

<code xml>
<?xml version="1.0" encoding="utf-8" ?>
<Batch
    xmlns="http://schemas.dbshell.com/core"
>
  <CopyTable>
    <CopyTable.Source>
      <CsvFile Name="c:/test/genre.csv" />
    </CopyTable.Source>
    <CopyTable.Target>
      <CreateTable Name="GenreFromFile" />
    </CopyTable.Target>
  </CopyTable>
</Batch>
</code>
