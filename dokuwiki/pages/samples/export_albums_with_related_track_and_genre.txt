====== Export Albums with related Track and Genre ======

This quite example exports object graph of consisting of Album, Track, Genre and MediaType.

<code xml>
<?xml version="1.0" encoding="utf-8" ?>
<Batch
    xmlns="http://schemas.dbshell.com/core"
    xmlns:dbmouse="http://schemas.dbmouse.com/dbshell"
    xmlns:ds="http://schemas.dbshell.com/dataset"
>
	<ds:DataSet />
 
	<ds:AddRows Table='Album' Condition="AlbumId&lt;5" />
	<ds:LoadReference Table='Track' Column='AlbumId' RefTable='Album' />
	<ds:LoadMissing Table='Genre' />
	
	<ds:KeepKey Table='MediaType' />
 
	<ds:Prepare />
 
	<ds:WriteSql File='ds1.sql' />	
</Batch>
</code>
