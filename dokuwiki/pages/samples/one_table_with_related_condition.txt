====== One table with related condition ======

This script gets all tracks, wgere genre name is "Pop".

<code xml>
<?xml version="1.0" encoding="utf-8" ?>
<Batch
    xmlns="http://schemas.dbshell.com/core"
    xmlns:ds="http://schemas.dbshell.com/dataset"
>
	<ds:DataSet />
 
	<ds:AddRows Table='Track' Condition=" {GenreId}.[Name]='Pop'" />
	<ds:KeepKey Table='Genre' />
 
	<ds:Prepare />
 
	<ds:WriteSql File='ds1.sql' />	
</Batch>
</code>
